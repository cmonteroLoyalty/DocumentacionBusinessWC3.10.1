Access Control

A permission is the combination of a business function and a single privilege granted to a user on a given resource type. Privileges are enforced by Digital Banking Services. Widget behavior is affected by the permissions of the user accessing the widget.

The following table describes how the widget behavior changes depending on the permissions of the user:

Resource

Business function

Privilege

Widget behavior

Contacts

Contacts

edit

The current user is able to edit a contact from the list of saved contacts.

If the user does not have the permission to edit, an error occurs when a submission is made.

Extension slots

This widget has the following extension slots:

  1. bbContactEditCustomizable

  2. bbContactEditFormCustomizable

  3. bbContactEditAddAccountCustomizable

bbContactEditCustomizable

Configure this extension slot using:

<ng-template bbContactEditCustomizable let-hostRef>
  <bb-header-ui
    headingType="h2"
    heading="Edit contact"
    i18n-heading="@@contact-edit.heading.label"
    role="heading"
    headingClasses="bb-block bb-block--lg"
  >
  </bb-header-ui>
  <hr>
  <div *ngIf="hostRef.error">
    <bb-alert-ui [dismissible]="true" title="" modifier="error" (close)="hostRef.error = null">
      <ng-container *ngIf="hostRef.error.message; else defaultError">{{hostRef.error.message}}</ng-container>
      <ng-template #defaultError>
        <ng-container i18n="@@contact-edit.error.default.message">Unexpected error has occured</ng-container>
      </ng-template>
    </bb-alert-ui>
  </div>
  <bb-contact-edit-form
    [contact]="hostRef.contact"
    (cancel)="hostRef.onCancel()"
    (save)="hostRef.onSave($event)"
  ></bb-contact-edit-form>
</ng-template>

bbContactEditFormCustomizable

Configure this extension slot using:

<ng-template bbContactEditFormCustomizable let-hostRef>
  <form name="editContactForm" #formRef="ngForm">
    <div class="bb-block bb-block--md">
      <bb-contact-personal-data
        [contact]="hostRef.formModel.contact.data"
        [address]="hostRef.contactAddress"
        [viewType]="hostRef.contactViewType"
        [isAddContactAddress]= "hostRef.isAddContactAddress"
        [submitted]="hostRef.isFormSubmitted"
        (contactAddressAdd)= "hostRef.addContactAddress($event)"
        (viewChange)="hostRef.changeContactView($event)">
      </bb-contact-personal-data>
    </div>
    <div class="bb-block bb-block--lg bb-stack">
      <bb-header-ui
        headingType="h2"
        heading="Account Information"
        i18n-heading="@@contact-edit.form.account-information.title"
        role="heading"
        headingClasses="bb-stack__item"
      >
      </bb-header-ui>
      <span
        class="bb-stack__item bb-stack__item--push-right small bb-text-support"
        i18n="@@contact-edit.form.account-information.accounts-number"
      >
        {hostRef.formModel.accounts.length, plural, =1 {1 account} other {{{hostRef.formModel.accounts.length}} accounts}}
      </span>
    </div>
    <div
      data-role="edit-contact-account-info"
      class="bb-block bb-block--md"
      *ngFor="let account of hostRef.formModel.accounts; let index = index; trackBy: hostRef.trackByIndex"
    >
      <div
        class="bb-stack"
        *ngIf="account.viewType !== 'CARD'"
      >
        <div class="bb-text-support bb-stack__item bb-stack__item--align-top bb-stack__item--push-right">
          <button
            [attr.data-role]="'create-contact-form-cancel-button'"
            bbButton
            buttonSize="sm"
            circle="true"
            color="link-dark"
            (click)="hostRef.deleteAccount(index)"
            bbTooltip="Discard account"
            triggers="hover focus"
            i18n-bbTooltip="@@contact-edit.form.account.button.discard.label"
          >
              <bb-icon-ui name="delete"></bb-icon-ui>
          </button>
        </div>
      </div>
      <div class="bb-block bb-block--lg">
        <bb-contact-account-data
          [index]="index"
          [account]="account.data"
          [accountView]="account.viewType"
          [bankAddressViewType]="account.bankAddressViewType"
          [accountHolderAddressViewType]="account.accountHolderAddressViewType"
          [isAddBankAddress]="account.isAddBankAddress"
          [canCopyAddress]="hostRef.onCanCopyAddress()"
          [isContactAddressCopied]="account.isContactAddressCopied"
          [bankAddress]="account.bankAddress"
          [accountHolderAddress]="account.accountHolderAddress"
          [isAddAccountHolderAddress]="account.isAddAccountHolderAddress"
          [submitted]="hostRef.isFormSubmitted"
          (bankAddressAdd)="hostRef.onAddBankAddress($event, index)"
          (accountHolderAddressAdd)="hostRef.addAccountHolderAddress($event, index)"
          (bankAddressViewChange)="hostRef.changeBankAddressView($event, index)"
          (accountHolderAddressViewChange)="hostRef.changeAccountHolderAddressView($event, index)"
          (editAccount)="hostRef.editAccount($event)"
          (deleteAccount)="hostRef.deleteAccount($event)"
          (contactAddressCopy)="hostRef.copyAddress($event, index)"
          (addressEditAfterCopy)="hostRef.onAddressEditAfterCopy(index)"
        >
        </bb-contact-account-data>
      </div>
    </div>

    <ng-container *ngIf="hostRef.formModel.accounts.length === 0">
      <div
        *ngIf="!hostRef.isFormSubmitted"
        role="alert"
        i18n="@@contact-edit.form.accounts.empty.message"
      >
        You need at least one account to save the contact modifications
      </div>
      <bb-alert-ui
        *ngIf="hostRef.isFormSubmitted"
        [dismissible]="false"
        title=""
        modifier="error"
        class="bb-inline-stack"
      >
        <span
          class="bb-inline-stack__item"
          data-role="edit-contact-no-account-error-message"
          i18n="@@contact-edit.form.accounts.empty.message"
        >
          You need at least one account to save the contact modifications
        </span>
      </bb-alert-ui>
    </ng-container>

    <div>
      <button
        bbButton
        color="link"
        data-role="add-additional-account"
        (click)="hostRef.isAddAccountOpen = true"
      >
        <bb-icon-ui name="add"></bb-icon-ui>
        <ng-container i18n="@@contact-edit.form.button.add.account">
          {hostRef.formModel.accounts.length, plural,
          =0 {Add account}
          other {Add additional account}}
        </ng-container>
      </button>
      <bb-contact-edit-add-account
        [isOpen]="hostRef.isAddAccountOpen"
        [index]="hostRef.formModel.accounts.length"
        [contactAddress]="hostRef.contactAddress"
        (confirm)="hostRef.addAccount($event)"
        (close)="hostRef.isAddAccountOpen = false"
      ></bb-contact-edit-add-account>
    </div>
    <div class="bb-button-bar">
      <button
        bbButton
        class="bb-button-bar__button"
        color="primary"
        data-role="contact-save-button"
        (click)="hostRef.onSubmit(formRef)"
        [disabled]="hostRef.isFormUnchanged(formRef)"
      >
        <ng-container i18n="@@contact-edit.form.button.submit">
            Save
        </ng-container>
      </button>
      <button
        class="bb-button-bar__button"
        bbButton
        color="secondary"
        data-role="contact-cancel-button"
        (click)="hostRef.onCancel(formRef)"
      >
        <ng-container i18n="@@contact-edit.form.button.cancel">
            Cancel
        </ng-container>
      </button>
    </div>

    <bb-confirm-action-dialog
      [isOpen]="hostRef.isCancelEditConfirmOpened"
      [confirmButtonColor]="'danger'"
      (confirm)="hostRef.isCancelEditConfirmOpened = false; hostRef.exit()"
      (cancel)="hostRef.isCancelEditConfirmOpened = false"
    >
      <span title i18n="@@contact-edit.confirm-action-dialog.title">Discard entered data?</span>
      <span body i18n="@@contact-edit.confirm-action-dialog.body">Are you sure to exit edition of the contact? The data you have entered will be lost.</span>
      <span cancelButtonLabel i18n="@@contact-edit.confirm-action-dialog.button.cancel">Not now</span>
      <span confirmButtonLabel i18n="@@contact-edit.confirm-action-dialog.button.confirm">Discard</span>
    </bb-confirm-action-dialog>
  </form>
</ng-template>
