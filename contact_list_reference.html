Access Control

A permission is the combination of a business function and a single privilege granted to a user on a given resource type. Privileges are enforced by Digital Banking Services. Widget behavior is affected by the permissions of the user accessing the widget.

The following table describes how the widget behavior changes depending on the permissions of the user:

Resource

Business function

Privilege

Widget behavior

Contacts

Contacts

view

The current user views a list of saved contacts.

If the user does not have the permission to view, the widget shows an empty list.

Extension slots

This widget has the following extension slots:

  1. bbContactListContainerCustomizable

  2. bbContactSearchCustomizable

  3. bbContactListCustomizable

  4. bbContactListItemCustomizable

bbContactListContainerCustomizable

This extension slot includes the following default template:

<ng-template bbContactListContainerCustomizable let-hostRef>
  <div class="col-md-9 col-12" *ngIf="!(hostRef.error || hostRef.isEmpty)">
    <bb-contact-search
      class="d-block"
      [debounceTime]="hostRef.debounceTime$ | async"
      (searchContacts)="hostRef.onContactSearch($event)">
    </bb-contact-search>
  </div>
  <ng-container *ngIf="!hostRef.error; else errorState">
    <bb-contact-list
      *ngIf="(hostRef.items$ | async) as items; else loading"
      [contactGroups]="items.itemGroups"
      [selectedContact]="hostRef.selectedContact"
      [contactItems]="items.items"
      [totalContacts]="items.totalCount"
      [loadMoreMethod]="hostRef.loadMoreMethod"
      [searchText]="hostRef.searchTerm"
      [contactsPerPage]="hostRef.pageSize"
      [loadMoreFrom]="hostRef.loadMoreFrom"
      [maxNavPages]="hostRef.maxNavPages"
      [isSearching]="hostRef.isSearching"
      [isLoading]="hostRef.isLoadingContacts$ | async"
      (pageChange)="hostRef.onPageChange($event)"
      (loadMore)="hostRef.onLoadMore($event)"
      (contactSelect)="hostRef.onContactSelected($event)"
      ></bb-contact-list>
  </ng-container>

  <ng-template #errorState>
    <div class="bb-state-container">
      <bb-empty-state-ui
        data-role="contact-manager-error-state"
        i18n-title="@@contact-manager.error.state.title"
        title="Page couldn’t load"
        i18n-subtitle="@@contact-manager.error.state.subtitle"
        subtitle="Try to reload the page or contact the system administrator."
        iconSize="xxl"
        iconModifier="error">
      </bb-empty-state-ui>
    </div>
  </ng-template>

  <ng-template #loading>
    <div class="bb-state-container">
      <bb-loading-indicator-ui i18n-text="Loading text|Loading data label@@bb-contacts-list-loading-label" text="Loading data..."></bb-loading-indicator-ui>
    </div>
  </ng-template>
</ng-template>

bbContactListCustomizable

This extension slot includes the following default template:

<ng-template bbContactListCustomizable let-hostRef>
  <div
    *ngIf="hostRef.contactItems && hostRef.contactItems.length"
    class="bb-block bb-block--md"
  >
    <ul
      class="bb-grouped-list bb-grouped-list--density-md"
      data-role="existing-contact"
      role="list"
    >
      <ng-container *ngFor="let item of hostRef.contactItems; let i = index">
        <li
          *ngIf="isListHeading(item, i)"
          class="bb-grouped-list__group bb-text-support bb-text-bold"
          data-role="header-row"
        >
          {{ hostRef.latestSeparator }}
        </li>
        <li
          class="bb-grouped-list__item"
          tabindex="0"
          role="button"
          aria-pressed="false"
          data-role="list-row"
          [ngClass]="{
            'bb-grouped-list__item--is-active': selected === i,
            'bb-grouped-list__item--no-separator': hasNoSeparatorClass(i)
          }"
          (click)="onContactSelect(item, i)"
          (keyup.Space)="onContactSelect(item, i)"
          (keyup.Enter)="onContactSelect(item, i)"
          #contactItem
        >
          <bb-contact-list-item
            [contactItem]="item"
            [showAlias]="true"
          ></bb-contact-list-item>
        </li>
      </ng-container>
    </ul>
  </div>

  <div
    [ngSwitch]="hostRef.loadMoreMethod"
    class="bb-pagination"
  >
    <ng-container *ngSwitchCase="'pagination'">
      <bb-pagination-ui
        [page]="0"
        [maxNavPages]="hostRef.maxNavPages"
        [totalItems]="hostRef.totalContacts"
        [itemsPerPage]="hostRef.contactsPerPage"
        (pageChange)="hostRef.onPageChange($event)"
        [boundaryLinks]="true"
        [directionLinks]="true"
      ></bb-pagination-ui>
    </ng-container>

    <ng-container *ngSwitchCase="'load-more'">
        <bb-load-button-ui
          class="bb-block bb-block--md"
          color="secondary"
          [isLoading]="hostRef.isLoading"
          [size]="hostRef.contactsPerPage"
          [from]="hostRef.loadMoreFrom"
          [totalCount]="hostRef.totalContacts"
          (paramChange)="hostRef.onLoadMore($event)">
          <ng-container  i18n="Load more contacts|Load more button for loading more contacts@@contact-manager-list.load.more">Load more</ng-container>
        </bb-load-button-ui>
    </ng-container>
  </div>

  <ng-template #searching>
    <div class="bb-state-container">
      <bb-loading-indicator-ui i18n-text="Searching text|Searching contacts label@@bb-contacts-list-searching-label" text="Searching..."></bb-loading-indicator-ui>
    </div>
  </ng-template>

  <ng-template #emptyState>
    <ng-container *ngIf="!hostRef.searchText; else emptySearchState">
      <div class="bb-state-container">
        <bb-empty-state-ui
          data-role="contact-manager-empty-state"
          i18n-title="@@contact-manager.state.nocontacts.title"
          title="No contacts"
          i18n-subtitle="@@contact-manager.state.nocontacts.subtitle"
          subtitle="You do not have any contacts to be displayed."
          iconSize="xxl"
          iconModifier="contacts">
        </bb-empty-state-ui>
      </div>
    </ng-container>
  </ng-template>

  <ng-template #emptySearchState>
    <div class="bb-state-container">
      <bb-empty-state-ui
        data-role="contact-manager-empty-state"
        i18n-title="@@contact-manager.state.no-search-results.title"
        title="No Results Found"
        i18n-subtitle="@@contact-manager.state.no-search-results.subtitle"
        subtitle="We can’t find what you’re looking for. Please try a different criteria."
        iconSize="xxl"
        iconModifier="search">
      </bb-empty-state-ui>
    </div>
  </ng-template>
</ng-template>
