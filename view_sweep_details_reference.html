View sweep details

Preferences

The widget has the following preferences that managers or administrators configure in Experience Manager:

Title and description Name Configurable for Default value

Input for sweep ID

The ID of the sweep selected to view details.

input.sweepId

Admin

route-param selectedSweepId (points to the ID from the widget URL)

Extension slots

This widget has the following extension slots:

  1. bbCashManagementSweepDetailsWidgetCustomizable

  2. bbCashManagementSweepDetailsModalCustomizable

  3. bbCashManagementSweepDetailsHeaderCustomizable

bbCashManagementSweepDetailsWidgetCustomizable

This extension slot includes the following default template:

<ng-container
  bbCashManagementSweepDetailsWidgetCustomizable
  [bbHostRef]="this"
  [bbTemplateContext]="{
    sweepId: selectedSweepId$ | async
  }"
>
</ng-container>

<ng-template bbCashManagementSweepDetailsWidgetCustomizable let-hostRef let-sweepId="context.sweepId">
  <bb-sweep-details-modal
    [sweepId]="sweepId"
    (sweepStatusChanged)="hostRef.onSweepStatusChanged($event)"
  ></bb-sweep-details-modal>
</ng-template>

bbCashManagementSweepDetailsModalCustomizable

This extension slot includes the following default template:

<ng-container
  bbCashManagementSweepDetailsModalCustomizable
  [bbHostRef]="this"
  [bbTemplateContext]="{
    sweep: sweep$ | async,
    error: error$ | async,
    isModalOpen: isModalOpen$ | async

></ng-container>

<ng-template
  bbCashManagementSweepDetailsModalCustomizable

  <bb-modal-ui
    [modalOptions]="hostRef.modalOptions"
    [isOpen]="isModalOpen"
    data-role="cash-management-create-sweep-modal"

      <bb-modal-header-ui data-role="cash-management-sweep-details-modal-window-header">

          <ng-template bbCustomModalHeader>

            <bb-cash-management-sweep-details-header
              *ngIf="sweep"

            >
            </bb-cash-management-sweep-details-header>
            <div class="d-flex ml-auto" id="no-print-button" >
              <bb-sweep-state-actions
                *ngIf="sweep"
                (click)="$event.stopImmediatePropagation()"
                (statusChange)="hostRef.onStatusChange($event)"
                [sweep]="sweep"
              >
                <ng-template let-actions="actions" let-sweep="sweep">
                  <bb-dropdown-menu-ui iconColor="dark" icon="ellipsis-h" btnColor="transparent" class="d-inline-block">
                    <ng-template bbDropdownMenuItem>
                      <a
                        *ngIf="hostRef.canBeCancelled(sweep)"
                        role="menuitem"
                        class="dropdown-item align"
                        data-role="cash-management-cancel-btn"
                        (bbKeyboardClick)="hostRef.canBeCancelled(sweep) && actions.cancel()"
                      >
                    <span i18n="Cancel sweep option@@cash-management.create-sweep.sweep-details.cancel-option">

                    </span>
                      </a>
                      <a
                        role="menuitem"
                        class="dropdown-item align"
                        data-role="cash-management-print-btn"

                    <span i18n="Print sweep option@@cash-management.create-sweep.sweep-details.print-option">
                      Print
                    </span>
                      </a>
                    </ng-template>
                  </bb-dropdown-menu-ui>
                </ng-template>
              </bb-sweep-state-actions>
              <button
                bbButton

                class="p-0 m-0"
                aria-label="Close"
                i18n-aria-label="Close modal button@@cash-management.create-sweep.create-sweep-modal.close-button"
                data-role="cash-management-create-sweep-modal-window-close-button"
                (click)="hostRef.closeModal()"
              >
                <bb-icon-ui name="times" color="dark"></bb-icon-ui>
              </button>
            </div>
          </ng-template>


      </bb-modal-header-ui>
      <bb-modal-body-ui>
        <ng-template bbCustomModalBody>
          <hr class="mb-4" />
          <ng-container *ngIf="sweep || error; else loadingState">
            <ng-container *ngIf="!error; else errorState">
              <bb-cash-management-sweep-details [sweep]="sweep"></bb-cash-management-sweep-details>

              <bb-sweep-state-actions [sweep]="sweep" (statusChange)="hostRef.onStatusChange($event)" id="no-print" >

                      data-role="cash-management-activate-btn"
                      (click)="actions.activate()"
                      bbCashManagementRenderWatcher
                      #activateBtn
                      (render)="handlers.onRender(activateBtn)"
                    >
                      <bb-icon-ui class="align-bottom" size="lg" name="play-arrow"></bb-icon-ui>

                        i18n="Activate button@@cash-management.sweep-details.activate-btn"
                        data-role="cash-management-sweep-status-btn-description"
                      >
                        Activate

                    </button>

                    <button

    <bb-common-error-state-ui [error]="error"></bb-common-error-state-ui>

    <bb-loading-state-ui
