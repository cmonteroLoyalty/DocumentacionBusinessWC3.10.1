Extension slots

The extension slots of this widget are:

  1. bbSelectContextContainerCustomizable

  2. bbSelectContextServiceAgreementItemCustomizable

  3. bbSelectContextServiceAgreementListCustomizable

bbSelectContextContainerCustomizable

This extension slot includes the following default template:

<ng-template
  bbSelectContextContainerCustomizable
  let-hostRef
  let-serviceAgreements="context.serviceAgreements"
  let-loading="context.loading"
  let-error="context.error"
  let-params="context.params"
  let-widgetProperties="context.widgetProperties"
  let-currentContext="context.currentContext"
  let-saving="context.saving"
>

  <div class="bb-select-context">
    <div class="bb-select-context__container">
      <ng-container *ngIf="serviceAgreements; else loadingTemplate">
        <bb-header-ui
          *ngIf="!saving"
          class="bb-block bb-block--md"
          headingType="h2"
          heading="Select context"
          data-role="select-context-header"
          i18n-heading="Select context|Select context label@@select-context.label.selectContext"
        ></bb-header-ui>

        <bb-alert-ui
          *ngIf="hostRef.savingError"
          title="Unknown error"
          i18n-title="Unknown Error|Unknown Error Label@@select-context.label.error.unknown">
          <span
            i18n="Try again|Try again error message@@select-context.label.error.tryAgain">
            Try to make action again or contact your system administrator.
          </span>
        </bb-alert-ui>

        <ng-container *ngIf="!saving && (hostRef.searchValue || serviceAgreements.totalCount > widgetProperties.showSearchAfter)">
          <bb-search-box-ui
            class="bb-block bb-block--lg"
            searchLabel="Search button"
            i18n-searchLabel="Search button|Search button@@select-context.label.searchLabel"
            showClear="true"
            showSearch="true"
            placeholder="Search"
            i18n-placeholder="Search|Search placeholder@@select-context.label.search"
            aria-label="Search service agreements"
            i18n-aria-label="Search service agreements|Search service agreements@@select-context.aria-label.search"
            data-role="Search"
            (submit)="hostRef.search($event)"
            (clear)="hostRef.search('')"
          ></bb-search-box-ui>
        </ng-container>

        <ng-container *ngIf="!error; else errorTemplate">
          <ng-container *ngIf="!(loading && params.from === 0); else loadingTemplate">
            <ng-container *ngIf="!saving; else savingTemplate">
              <ng-container *ngIf="serviceAgreements.items.length > 0; else emptyTemplate">
                <bb-infinite-scroll-ui
                  class="bb-infinite-scroll"
                  [offset]="5"
                  [disableScrollEnd]="loading || serviceAgreements.items.length === serviceAgreements.totalCount"
                  (scrollEnd)="loadMore()">
                  <bb-service-agreement-list
                    [serviceAgreements]="serviceAgreements.items"
                    [loading]="loading"
                    [currentContext]="currentContext"
                    [searchValue]="hostRef.searchValue"
                    [totalCount]="serviceAgreements.totalCount"
                    [from]="params.from"
                    (loadMore)="hostRef.loadMore()"
                    (select)="hostRef.select($event)"
                  ></bb-service-agreement-list>
                </bb-infinite-scroll-ui>
              </ng-container>
            </ng-container>
          </ng-container>
        </ng-container>

        <ng-container *ngIf="widgetProperties.backOutput && !saving">
          <div class="bb-block bb-block--lg"></div>
          <div class="d-block">
            <div class="bb-button-bar bb-button-bar--reverse">
              <button bbButton
                      (click)="hostRef.goBack()"
                      color="secondary"
                      class="bb-button-bar__button"
                      data-role="select-context-back-button"
              >
                <span i18n="Back button|Button to navigate back@@select-context.label.backButton">
                  Back
                </span>
              </button>
            </div>
          </div>
        </ng-container>
      </ng-container>
    </div>
  </div>

  <ng-template #loadingTemplate>
    <div class="bb-state-container bb-select-context__state">
      <bb-loading-indicator-ui
      ></bb-loading-indicator-ui>
    </div>
  </ng-template>

  <ng-template #savingTemplate>
    <div class="bb-state-container bb-select-context__state">
      <bb-loading-indicator-ui
        text="Processing the request..."
        i18n-text="Processing the request|Text to show while saving the selected context@@select-context.label.processing"
      ></bb-loading-indicator-ui>
    </div>
  </ng-template>

  <ng-template #errorTemplate>
    <div class="bb-state-container bb-select-context__state">
      <bb-select-context-error
        [errorKind]="error?.kind"
      ></bb-select-context-error>
    </div>
  </ng-template>

  <ng-template #emptyTemplate>
    <div class="bb-state-container bb-select-context__state">
      <bb-empty-state-ui
        *ngIf="hostRef.searchValue"
        title="No Results Found"
        i18n-title="Search empty state title|Empty state label when there is no search results@@select-context.search.empty.title"
        subtitle="We can’t find what you’re looking for. Please try a different criteria."
        i18n-subtitle="Search empty state subtitle|Empty state subtitle when there is no search results@@select-context.search.empty.subtitle"
        iconModifier="search"
        iconColor="muted"
        iconSize="xxl"
      ></bb-empty-state-ui>

      <bb-empty-state-ui
        *ngIf="!hostRef.searchValue"
        title="Permissions required."
        i18n-title="Permissions required|Empty state label when user has no permissions@@select-context.permissions.empty.title"
        subtitle="Please contact the administrator to activate your account."
        i18n-subtitle="Permissions required subtilte|Empty state subtitle when user has no permissions@@select-context.permissions.empty.subtitle"
        iconModifier="lock"
        iconColor="muted"
        iconSize="xxl"
      ></bb-empty-state-ui>
    </div>
  </ng-template>

</ng-template>

bbSelectContextServiceAgreementItemCustomizable

This extension slot includes the following default template:

<ng-template
  bbSelectContextServiceAgreementItemCustomizable
  let-hostRef
  let-item="context">

  <div class="bb-stack bb-stack--align-top">
    <div class="bb-stack__item">
      <bb-user-context-icon-ui
        [isMaster]="item?.isMaster"
        [name]="item?.name"
      ></bb-user-context-icon-ui>
    </div>
    <div class="bb-stack__item bb-stack__item--fill">
      <div class="bb-text-bold" data-role="service-agreement-item-name">{{item?.name}}</div>
      <div class="bb-subtitle bb-text-support" data-role="service-agreement-item-description">{{item?.description}}</div>
    </div>
  </div>

</ng-template>

bbSelectContextServiceAgreementListCustomizable

This extension slot includes the following default template:

<ng-template
  bbSelectContextServiceAgreementListCustomizable
  let-hostRef
  let-serviceAgreements="context.serviceAgreements"
  let-currentContext="context.currentContext"
  let-loading="context.loading"
  let-searchValue="context.searchValue"
  let-totalCount="context.totalCount"
  let-from="context.from">

  <ng-container *ngIf="currentContext && !searchValue">
    <div
      data-role="current-context-label"
      class="bb-select-context__subheader"
      i18n="Current context|Current context label@@select-context.label.currentContext">
      Current context
    </div>
    <div class="list-group list-group-flush">
      <div
        role="button"
        data-role="current-context-data"
        tabindex="0"
        (click)="hostRef.onSelect(currentContext)"
        (keyup.enter)="hostRef.onSelect(currentContext)"
        class="bb-select-context__current-context list-group-item list-group-item-action"
      >
        <bb-service-agreement-item
          [item]="currentContext"
        ></bb-service-agreement-item>
      </div>
    </div>

    <div class="bb-select-context__subheader"
         data-role="switch-context-label"
      i18n="Switch to context|Switch to context label@@select-context.label.switchToContext">
      Switch to context
    </div>
  </ng-container>

  <div class="bb-select-context__subheader" *ngIf="searchValue && totalCount > 0">
  <span i18n="Results found|Results found label@@select-context.label.resultsFound">
    {totalCount, plural, =1 {1 result found} other {{{totalCount}} results found}}
  </span>
  </div>

  <div class="list-group list-group-flush">
    <div *ngFor="let item of serviceAgreements; let first = first; let last = last"
      role="button"
      tabindex="0"
      [attr.data-role]="'service-agreement-list-item-' + item.name"
      (click)="hostRef.onSelect(item)"
      (keyup.enter)="hostRef.onSelect(item)"
      [class.first]="first"
      [class.last]="last"
      class="list-group-item list-group-item-action bb-select-context__item">
      <bb-service-agreement-item
        [item]="item"
      ></bb-service-agreement-item>
    </div>
  </div>

  <div [class.invisible]="!loading" *ngIf="serviceAgreements.length !== totalCount">
    <bb-loading-indicator-ui loaderSize="sm"></bb-loading-indicator-ui>
  </div>

  <div class="bb-select-context__items-loaded-label" *ngIf="from > 0 && serviceAgreements.length === totalCount">
    <span *ngIf="!searchValue"
      i18n="All items loaded|All items loaded label@@select-context.label.allItemsLoaded">
      All items are loaded
    </span>
    <span *ngIf="searchValue"
      i18n="All items loaded|All items loaded label@@select-context.label.allResultsLoaded">
      All results are loaded
    </span>
  </div>
</ng-template>
