Manage sweeps

Preferences

This widget has the following preferences that managers or administrators configure in Experience Manager:

Title and description Name Configurable for Default value

Output for sweep ID

output.selectedSweepId

Admin

route-param selectedSweepId (points to the value from the widget URL)

Number of sweeps per page

pageSize

Admin

10

Extension slots

This widget has the following extension slots:

  1. bbCashManagementSweepsListCustomizable

  2. bbCashManagementSweepsListTableCustomizable

  3. bbCashManagementSweepsListItemCustomizable

  4. bbCashManagementSweepFilterCustomizable

bbCashManagementSweepsListCustomizable

This extension slot includes the following default template:

<ng-container
  bbCashManagementSweepListCustomizable
  [bbHostRef]="this"
  [bbTemplateContext]="{
    sweeps: sweeps$ | async,
    error: error$ | async,
    totalCount: sweepsTotalCount$ | async,
    currentPage: currentPage$ | async,
    pageSize: pageSize$ | async,
    isLoading: isLoading$ | async,
    isMobileView: isMobileView$ | async,
    searchParamsAvailable: searchParamsAvailable$ | async,
    loadMore: loadMore$ | async,
    hasMoreSweepsToLoad: hasMoreSweepsToLoad$ | async
  }"
></ng-container>

<ng-template
  bbCashManagementSweepListCustomizable
  let-hostRef
  let-sweeps="context.sweeps"
  let-error="context.error"
  let-totalCount="context.totalCount"
  let-currentPage="context.currentPage"
  let-pageSize="context.pageSize"
  let-isLoading="context.isLoading"
  let-isMobileView="context.isMobileView"
  let-searchParamsAvailable="context.searchParamsAvailable"
  let-loadMore="context.loadMore"
  let-hasMoreSweepsToLoad="context.hasMoreSweepsToLoad"
>
  <bb-cash-management-sweep-filter (sweepsSearch)="hostRef.onSweepsSearch($event)"></bb-cash-management-sweep-filter>

  <p
    *ngIf="sweeps?.length && searchParamsAvailable && !isLoading && !error"
    class="px-3 pt-3 mt-2 border-top font-weight-bold"
    data-role="sweeps-list-found-number"
    i18n="Total results found@@cash-management.sweeps.search.total-count"
  >
    {totalCount, plural, =1 {{{ totalCount }} result found} other {{{ totalCount }} results found}}
  </p>

  <ng-container *ngIf="sweeps || error; else searchState">
    <ng-container *ngIf="!isLoading || loadMore; else searchState">
      <ng-container *ngIf="!error; else errorState">
        <ng-container *ngIf="(sweeps.length || !searchParamsAvailable); else searchNoResults">
          <ng-container *ngIf="sweeps.length; else emptyState">
            <bb-cash-management-sweep-list-table
              [sweeps]="sweeps"
              (sweepSelected)="hostRef.onSweepSelected($event)"
              (sweepStatusChange)="hostRef.onSweepStatusChange($event)"
            ></bb-cash-management-sweep-list-table>
            <bb-pagination-ui
              *ngIf="!isMobileView"
              [page]="currentPage"
              [boundaryLinks]="true"
              [directionLinks]="true"
              [totalItems]="totalCount"
              [itemsPerPage]="pageSize"
              [maxNavPages]="4"
              (pageChange)="hostRef.onPageChange($event)"
            ></bb-pagination-ui>
            <div class="d-flex" *ngIf="isMobileView && hasMoreSweepsToLoad">
              <bb-load-button-ui
                i18n="Load more button@@cash-management.sweeping.list.load-more"
                class="mx-auto"
                color="secondary"
                [from]="currentPage"
                [size]="pageSize"
                [totalCount]="totalCount"
                [isLoading]="isLoading"
                (paramChange)="hostRef.onLoadMore()"
              >
                Load more
              </bb-load-button-ui>
            </div>
          </ng-container>
        </ng-container>
      </ng-container>
    </ng-container>
  </ng-container>

  <ng-template #emptyState>
    <div class="py-5 my-5">
      <bb-empty-state-ui
        i18n-title="Sweeping empty state message title@@cash-management.state.noSweeps.title"
        title="No Sweeps"
        iconModifier="vertical-align-center"
        i18n-subtitle="Sweeping empty state message@@cash-management.state.noSweeps.subtitle"
        subtitle="There are no sweeps to be displayed."
        iconSize="xxl"
      ></bb-empty-state-ui>
    </div>
  </ng-template>

  <ng-template #errorState>
    <div class="py-5 my-5">
      <bb-empty-state-ui
        i18n-title="Unknown error title@@error-state.error.unknown.title"
        title="Unknown Error"
        i18n-subtitle="Unknown error message@@error-state.error.unknown.message"
        subtitle="Try to reload the page or contact the system <br>administrator."
        iconSize="xxl"
        iconModifier="error-outline"
      ></bb-empty-state-ui>
    </div>
  </ng-template>

  <ng-template #searchState>
    <div
      class="d-flex justify-content-center align-items-center bb-cash-management-sweep-list-container"
      data-role="sweeps-list-search-loading"
    >
      <bb-loading-indicator-ui loaderSize="md"></bb-loading-indicator-ui>
    </div>
  </ng-template>

  <ng-template #searchNoResults>
    <div
      class="d-flex justify-content-center align-items-center bb-cash-management-sweep-list-container"
      data-role="sweeps-list-no-search-result"
    >
      <bb-empty-state-ui
        i18n-title="Sweep search no result title@@cash-management.sweeps.search.no-results.title"
        title="No Results Found"
        i18n-subtitle="Sweep search no result subtitle@@cash-management.sweeps.search.no-results.subtitle"
        subtitle="We can't find what you are looking for.<br> Please try a different criteria."
        iconSize="xxl"
        iconModifier="search"
        iconColor="dark"
      ></bb-empty-state-ui>
    </div>
  </ng-template>
</ng-template>
