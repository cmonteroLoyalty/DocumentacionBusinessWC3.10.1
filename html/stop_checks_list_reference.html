Extension slots

This widget has the following extension slots:

  1. bbStopChecksListCustomizableDirective

  2. bbStopChecksDetailsCustomizableDirective

  3. bbStopChecksEmptyListCustomizableDirective

bbStopChecksListCustomizableDirective

This extension slot includes the following default template:

<ng-template bbStopChecksListCustomizableDirective let-hostRef>
  <div class="table-responsive">
    <table
      class="table table-hover bb-table bb-block--no-margin"
      [bbTable]="hostRef.checks"
      bbTableFocus
    >
      <thead>
      <tr>
        <th data-role="stop-checks-header-account" scope="col">
            <span
              i18n="@@stop-checks.header.account"
            >
              Check account
            </span>
        </th>
        <th data-role="stop-checks-header-check-numbers" scope="col">
            <span
              i18n="@@stop-checks.header.numbers"
            >
              Check number(s)
            </span>
        </th>
        <th data-role="stop-checks-header-check-submitted-date" scope="col">
            <span
              i18n="@@stop-checks.header.date.submitted"
            >
              Date submitted
            </span>
        </th>
        <th
          scope="col"
          data-role="stop-checks-header-check-expiry-date"
          i18n="@@stop-checks.header.expiry.date"
        >
          <span>Expiry date</span>
        </th>
        <th
          scope="col"
          data-role="stop-checks-header-status"
          i18n="@@stop-checks.header.status"
        >
          <span>Status</span>
        </th>
      </tr>
      </thead>
      <tbody>
      <tr
        *ngFor="let check of hostRef.checks"
        data-role="stop-checks-row"
        [ngClass]="{ 'text-muted': check.status.toUpperCase() === 'EXPIRED' }"
        (keydown.enter)="hostRef.onPaymentItemClick(check)"
        (keydown.space)="hostRef.onPaymentItemClick(check)"
        (click)="hostRef.onPaymentItemClick(check)"
        tabindex="0"
      >
        <td class="d-none d-md-table-cell">
          <div data-role="stop-check-originator-account-name">{{ check.originatorAccount.name }}</div>
          <div class="bb-text-support"
               data-role="stop-check-originator-account-number">{{ check.originatorAccount.identification }}</div>
        </td>
        <td>
          <div class="d-block d-md-none bb-block--sm"></div>
          <div class="d-block d-md-none bb-block--sm">
            <bb-payord-status-badge [status]="check.status"></bb-payord-status-badge>
          </div>

          <div class="d-block d-md-none">
            <div class="bb-block--sm">
              <div class="bb-text-bold" i18n="@@stop-checks.header.small.account">
                Check account
              </div>
              <div data-role="stop-check-originator-account-name">{{ check.originatorAccount.name }}</div>
              <div class="bb-text-support"
                   data-role="stop-check-originator-account-number">{{ check.originatorAccount.identification }}</div>
            </div>

            <div class="bb-block--sm">
              <div class="bb-text-bold" i18n="@@stop-checks.header.small.numbers">
                Check number
              </div>
              <ng-container *ngIf="check.check">{{ check.check.number }}</ng-container>
              <ng-container *ngIf="check.checkNumberRange">
                {{ check.checkNumberRange.firstNumber }} <span
                i18n="@@stop-checks.table.label.to">to</span> {{ check.checkNumberRange.lastNumber }}
              </ng-container>
            </div>

            <div class="row">
              <div class="col-6">
                <div class="bb-text-bold" i18n="@@stop-checks.header.small.date.submitted">
                  Date submitted
                </div>
                <div>{{ check.dateOfRequest | date: 'MM/dd/y' }}</div>
              </div>
              <div class="col-6">
                <div class="bb-text-bold" i18n="@@stop-checks.header.small.date.expired">
                  Date expired
                </div>
                <div *ngIf="check.expirationDate; else noValue">{{ check.expirationDate | date: 'MM/dd/y' }}</div>
              </div>
            </div>
          </div>

          <div class="d-none d-md-block">
            <ng-container *ngIf="check.check">{{ check.check.number }}</ng-container>
            <ng-container *ngIf="check.checkNumberRange">
              {{ check.checkNumberRange.firstNumber }} <span
              i18n="@@stop-checks.table.label.to">to</span> {{ check.checkNumberRange.lastNumber }}
            </ng-container>
          </div>
        </td>
        <td class="d-none d-md-table-cell" data-role="stop-check-col-date">
          {{ check.dateOfRequest | date: 'MM/dd/y' }}
        </td>
        <td class="d-none d-md-table-cell" data-role="stop-check-col-date">
          <span *ngIf="check.expirationDate; else noValue">{{ check.expirationDate | date: 'MM/dd/y' }}</span>
        </td>
        <td class="d-none d-md-table-cell" data-role="stop-check-col-status">
          <bb-payord-status-badge [status]="check.status"></bb-payord-status-badge>
        </td>
      </tr>
      </tbody>
    </table>
  </div>
</ng-template>

<ng-template
  bbStopChecksDetailsCustomizableDirective
  let-hostRef
>
  <bb-modal-ui
    [isOpen]="hostRef.isOpen"
    (cancel)="hostRef.onClose()"
    (confirm)="hostRef.onClose()"
  >
    <ng-container *ngIf="hostRef.check; else loading">
      <bb-modal-header-ui>
        <ng-template bbCustomModalHeader>
          <div class="bb-stack" data-role="stop-checks-details-header">
            <div class="bb-stack__item">
              <bb-header-ui
                headingType="h2"
                heading="Stop check request"
                i18n-heading="@@stop-checks.details.header.title"
              ></bb-header-ui>
              <bb-payord-status-badge [status]="hostRef.check.status"></bb-payord-status-badge>
              <span class="text-small bb-text-support">
                <span i18n="@@stop-checks.details.submitted.on"> Submitted on </span>
                <span>{{ hostRef.check.dateOfRequest | date: 'MM/dd/y' }}</span>
              </span>
            </div>

            <div class="bb-stack__item bb-stack__item--push-right bb-stack__item--align-top d-print-none">
              <div class="bb-stack">
                <div class="bb-stack__item">
                  <button
                    bbButton
                    [circle]="true"
                    buttonSize="sm"
                    color="link-dark"
                    (click)="hostRef.onPrint()"
                    (keydown.enter)="hostRef.onPrint()"
                    type="button"
                    aria-label="Print stop check detail dialog"
                    data-role="print-stop-checks-details-modal"
                    i18n-aria-label="@@stop-checks.check.header.print"

                  </button>
                </div>
                <div class="bb-stack__item">
                  <button
                    bbButton
                    [circle]="true"
                    buttonSize="sm"
                    color="unstyled"
                    (click)="hostRef.onClose()"
                    (keydown.enter)="hostRef.onClose()"
                    aria-label="Close stop check detail dialog"
                    data-role="close-stop-checks-details-modal"
                    i18n-aria-label="@@stop-checks.stop-checks.header.close"
                  >
                    <bb-icon-ui color="muted" name="clear"></bb-icon-ui>
                  </button>
                </div>
              </div>
            </div>
          </div>
        </ng-template>
      </bb-modal-header-ui>

      <bb-modal-body-ui>
        <ng-template bbCustomModalBody>
          <div class="bb-block--md">
            <bb-header-ui
              headingType="h4"
              heading="Check details"
              i18n-heading="@@stop-checks.details.body.title"
            ></bb-header-ui>
          </div>


          <div class="bb-block--md">
            <div class="bb-label bb-block--xs" i18n="@@stop-checks.details.body.account.label">Check Account</div>
            <div>{{ hostRef.check.originatorAccount.name }}</div>
            <div class="bb-text-support">{{ hostRef.check.originatorAccount.identification }}</div>
          </div>

          <ng-container *ngIf="hostRef.check.check; else numbersRange">
            <div class="bb-stack row break-word">
              <div class="bb-stack__item col-12 bb-block--md" [ngClass]="{ 'col-md-6': hostRef.check.check.amount }">
                <div
                  class="bb-label bb-block--xs"
                  i18n="@@stop-checks.details.body.check.number"
                >Check number</div>
                <div>{{ hostRef.check.check.number }}</div>
              </div>
              <div *ngIf="hostRef.check.check.amount" class="bb-stack__item col-12 col-md-6 bb-block--md">
                <div
                  class="bb-label bb-block--xs"
                  i18n="@@stop-checks.details.body.check.amount"
                >Check amount</div>
                <bb-amount-ui
                  [amount]="hostRef.check.check.amount?.amount"
                  [currency]="hostRef.check.check.amount?.currencyCode"
                ></bb-amount-ui>
              </div>
            </div>

            <div class="bb-stack row break-word">
              <div
                *ngIf="hostRef.check.check.dateIssued"
                class="bb-stack__item col-12 bb-block--md"
                [ngClass]="{ 'col-md-6': hostRef.check.check.payeeName }"
              >
                <div
                  class="bb-label bb-block--xs"
                  i18n="@@stop-checks.details.body.check.date"
                >Check date</div>
                <div>{{ hostRef.check.check.dateIssued | date: 'MM/dd/y' }}</div>
              </div>
              <div *ngIf="hostRef.check.check.payeeName" class="bb-stack__item col-12 col-md-6 bb-block--md">
                <div
                  class="bb-label bb-block--xs"
                  i18n="@@stop-checks.details.body.check.payee.name"
                >Payee name</div>
                <div>{{ hostRef.check.check.payeeName }}</div>
              </div>
            </div>
          </ng-container>

          <div class="bb-stack row break-word">
            <div
              *ngIf="hostRef.check.reason"
              class="bb-stack__item"
              [ngClass]="{ 'col-md-6': hostRef.check.expirationDate }"
            >
              <div
                class="bb-label bb-block--xs"
                i18n="@@stop-checks.details.body.stop.reason"
              >Stop reason</div>
              <div>{{ hostRef.check.reason }}</div>
            </div>
            <div class="d-md-none">&nbsp;</div>
            <div *ngIf="hostRef.check.expirationDate" class="bb-stack__item col-md-6">
              <div
                class="bb-label bb-block--xs"
                i18n="@@stop-checks.details.body.expiration.date"
              >Stop expiration date</div>
              <div>{{ hostRef.check.expirationDate | date: 'MM/dd/y' }}</div>
            </div>
          </div>
        </ng-template>
      </bb-modal-body-ui>
    </ng-container>

    <ng-template #numbersRange>
      <div class="bb-stack row break-word">
        <div class="bb-stack__item col-12 col-md-6 bb-block--md">
          <div
            class="bb-label bb-block--xs"
            i18n="@@stop-checks.details.body.range.first.label"
          >First check number</div>
          <div>{{ hostRef.check.checkNumberRange.firstNumber }}</div>
        </div>
        <div class="bb-stack__item col-12 col-md-6 bb-block--md">
          <div
            class="bb-label bb-block--xs"
            i18n="@@stop-checks.details.body.range.last.label"
          >Last check number</div>
          <div>{{ hostRef.check.checkNumberRange.lastNumber }}</div>
        </div>
      </div>
    </ng-template>

    <ng-template #loading>
      <bb-loading-indicator-ui></bb-loading-indicator-ui>
    </ng-template>
  </bb-modal-ui>
</ng-template>
