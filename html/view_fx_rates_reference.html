View FX rates

Preferences

The widget has the following preferences that managers or administrators configure in Experience Manager:

Name

Description

Configurable for

Default value

pageSize

Sets the number of currency rates shown per page.

Admin

5

onlyFavorite

Shows only favorite currency pairs.

Admin

false

visibilityOfPairInfo

Shows the details of a currency pair when clicked.

Admin

false

typeOfCalculation

Sets the value to be used in the calculation for the candlestick chart. Set it to Ask/Bid, None, or Open/Close. This preference is available only when visibilityOfPairInfo is set to true.

Admin

None

autoUpdateInterval

Sets the interval to poll the server for new currency rates, in seconds.

Admin

15

output.selectedCurrenciesPair

The widget output once the customer clicks Place order.

Admin

navigation:

output.pairDetailsId

The widget output once the customer selects a currency pair.

Admin

navigation:

output.fxRatesSearch

The widget output for the search results.

Admin

route

Extension slots

This widget has the following extension slots:

  1. bbFxRatesTableWidgetCustomizable: Desktop and mobile view

  2. bbFxRatesTableCustomizable: Desktop view

  3. bbFxRatesTableItemCustomizable: Desktop view

  4. bbFxCurrencyPairCustomizable: Desktop view

  5. bbFxRatesAdaptiveCandleChartCustomizable: Desktop view

  6. bbFxRatesSearchWidgetCustomizable: Desktop view

  7. bbFxRatesListCustomizable: Mobile view

  8. bbFxRatesListItemCustomizable: Mobile view

  9. bbFxRatesFavoriteButtonCustomizable: Desktop and mobile view

bbFxRatesTableWidgetCustomizable

This extension slot includes the following default template:

<ng-container
  bbFxRatesTableWidgetCustomizable
  [bbHostRef]="this"
  [bbTemplateContext]="{
    currenciesRates: currenciesRates$ | async,
    isLoading: isLoading$ | async,
    isMobileView: isMobileView$ | async,
    totalCount: totalCount$ | async,
    currentPage: currentPage$ | async,
    pageSize: pageSize$ | async,
    error: error$ | async,
    loadMore: loadMore$ | async,
    searchParamsAvailable: searchParamsAvailable$ | async,
    showError: showError$ | async,
    loadingState: loadingState$ | async,
    pairDetail: pairDetails$ | async,
    typeOfCalculation: typeOfCalculation$ | async,
    visibilityOfPairInfo: visibilityOfPairInfo$ | async,
    hasMore: hasMore$ | async
  }"
></ng-container>

<ng-template
  bbFxRatesTableWidgetCustomizable
  let-hostRef
  let-currenciesRates="context.currenciesRates"
  let-isLoading="context.isLoading"
  let-isMobileView="context.isMobileView"
  let-totalCount="context.totalCount"
  let-currentPage="context.currentPage"
  let-pageSize="context.pageSize"
  let-error="context.error"
  let-loadMore="context.loadMore"
  let-searchParamsAvailable="context.searchParamsAvailable"
  let-showError="context.showError"
  let-loadingStateDetail="context.loadingState"

  let-hasMore="context.hasMore"
>

    <ng-container *ngIf="!isLoading || loadMore; else loadingState">
      <ng-container *ngIf="!error; else errorState">

          <div class="row" #tableContainer>
            <div class="col bb-fx-rates-table-wrapper" >
          <p
            *ngIf="searchParamsAvailable"
            class="d-flex px-3 pt-3 border-top font-weight-bold"
            data-role="fx-rates-table-found-number"
            i18n="Total results found@@fx-trading.search.total-count"
          >
            {totalCount, plural, =1 {{{ totalCount }} result found} other {{{ totalCount }} results found}}
          </p>

          <ng-container *ngIf="isMobileView; else fxRatesTable">

              <bb-load-button-ui
                i18n="Load more button@@fx-trading.table.load-more"
                class="mx-auto"
                color="secondary"
                data-role="fx-trading-load-more-btn"
                *ngIf="loadMore || hasMore"
                [from]="currentPage"
                [size]="pageSize"
                [totalCount]="totalCount"
                [isLoading]="isLoading"
                (paramChange)="hostRef.onLoadMore()"

                Load more
              </bb-load-button-ui>
            </div>
          </ng-container>

          <ng-template #fxRatesTable>
                <bb-fx-rates-table
                  [currenciesRates]="currenciesRates"

                  (pairKey)="hostRef.onPairKeyChanged($event)"
                  [selectedPairKey]="hostRef.pairKeySelected"
                  [visibilityOfPairInfo]="visibilityOfPairInfo"
                  [tableContainerWidth]="tableContainer.clientWidth"

                ></bb-fx-rates-table>
                <bb-pagination-ui
                  [page]="currentPage"
                  [boundaryLinks]="tableContainer.clientWidth <= 700 ? false : true"
                  [directionLinks]="true"
                  [totalItems]="totalCount"
                  [itemsPerPage]="pageSize"
                  [maxNavPages]="4"
                  (pageChange)="hostRef.onPageChange($event)"
                ></bb-pagination-ui>
              </ng-template>
            </div>
          <div
            data-role="fx-rates-pair-detail-info"
            [ngClass]="visibilityOfPairInfo ? 'mr-3  pl-3  bb-fx-rates-pair-info d-none  d-md-block' : 'd-none'"
            style="margin-top: -7.5rem; margin-bottom: -1.75rem"
            *ngIf="hostRef.pairKeySelected && !isMobileView"
          >
            <ng-container *ngIf="loadingStateDetail !== null; else loading">
              <ng-container *ngIf="!showError; else errorStateDetail">
                <ng-container *ngIf="pairDetail">
                  <bb-fx-rates-pair-info
                    (closeEmit)="hostRef.onCloseDetail()"
                    (tradeOrderEmit)="hostRef.openTradeOrder(pairDetail)"
                    (detailPageEmit)="hostRef.openDetailPage(pairDetail)"
                    [pairKeyInfo]="pairDetail"
                    [typeOfCalculation]="typeOfCalculation"
                  >
                  </bb-fx-rates-pair-info>
                </ng-container>
              </ng-container>
            </ng-container>
            </div>
          </div>
        </ng-container>

      </ng-container>
    </ng-container>
  </ng-container>


  <ng-template #errorStateDetail>
    <div class="bb-fx-rates-pair-detail-container d-flex justify-content-center align-items-center">
      <bb-error-state-ui

        iconModifier="error"

    </div>
  </ng-template>

  <ng-template #emptyState>

      class="d-flex justify-content-center align-items-center bb-fx-rates-table-container"
      data-role="fxrates-loading"

      <bb-empty-state-ui

  </ng-template>

      class="d-flex justify-content-center align-items-center bb-fx-rates-table-container"

    </div>
