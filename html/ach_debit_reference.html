Initiate ACH debit

Preferences

The widget has the following preferences that managers or administrators edit in Experience Manager:

Label Name Value type Default value

Create a debit event trigger

createDebitEventTrigger

String

bb.event.initiate.create.debit

Flag for debit modal

isDebitModalView

String

false

Extension slots

This widget has the following extension slots:

  1. bbPayordDebitCreateContainerCustomizable

  2. bbPayordDebitCreateModalCustomizable

bbPayordDebitCreateContainerCustomizable

This extension slot includes the following default template:

<ng-container
  bbPayordDebitCreateContainerCustomizable
  [bbHostRef]="hostRef"
></ng-container>
<ng-template
  bbPayordDebitCreateContainerCustomizable
  let-hostRef
>
  <ng-template
    [ngIf]="hostRef.isDebitModalView$ | async"
    [ngIfElse]="pageView"
  >
    <bb-payord-debit-create-modal
      #modalView
      [createDebitEventTrigger]="hostRef.createDebitEventTrigger$ | async"
    >
      <ng-container bbPayordDebitCreateModalBody>
        <ng-container *ngIf="(hostRef.state$ | async) === hostRef.states.Edit; else preview">
          <bb-header-ui
            headingType="h2"
            *ngIf="!(modalView.debitItemForCopy$ | async)"
            heading="New ACH Debit"
            data-role="payord-debit-create-edit-header"
            i18n-heading="New Ach Debit title|Title of the Create new ach debit page@@payord-debit-create.edit.title"
          ></bb-header-ui>
          <bb-header-ui
            *ngIf="modalView.debitItemForCopy$ | async"
            headingType="h2"
            heading="Copy Debit"
            data-role="payord-debit-create-edit-header"
            i18n-heading="Copy Debit title|Title of the Create new ach debit page@@payord-debit-create.edit.copy.title"
          ></bb-header-ui>
          <bb-payord-debit-create-edit [form]="hostRef.form"></bb-payord-debit-create-edit>
        </ng-container>
      </ng-container>
      <ng-container bbPayordDebitCreateModalFooter>
        <ng-container *ngIf="(hostRef.state$ | async) === hostRef.states.Edit">

          <div class="bb-button-bar bb-button-bar--spacing-lg">

            <ng-container *ngTemplateOutlet="continueButton"></ng-container>

            <bb-button-ui
                class="bb-button-bar__button"
                (click)="hostRef.openModal()"
                color="secondary"
                data-role="ach-debit-preview-cancel-button"
                i18n="Button text|Cancel@@payord-debit-create.preview.cancel.button"
              >Cancel

      </ng-container>
    </bb-payord-debit-create-modal>

  </ng-template>
</ng-template>
