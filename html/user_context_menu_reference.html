Preferences

The widget has the following preferences that managers or administrators can edit in Experience Manager:

Title and description Name Admin Default value Data type

Number of service agreements to load per page

pageSize

7

long

Show search if more than service agreements

showSearchAfter

includeCurrentContext

Preferences that have the Admin column checked, are only editable for administrators.

Extension slots

The extension slots of this widget are:

  1. bbUserContextMenuContainerCustomizable

  2. bbUserContextMenuDropdownCustomizable

  3. bbUserContextMenuInfoCustomizable

For additional extension slots see Select context.

bbUserContextMenuContainerCustomizable

This extension slot includes the following default template:

<ng-template
  bbUserContextMenuContainerCustomizable
  let-hostRef
  let-currentContext="context.currentContext"
  let-canSelectContext="context.canSelectContext"
  let-userInfo="context.userInfo">

  <div class="user-context-dropdown">
    <bb-dropdown-menu-ui btnColor="unstyled" container="body" fullWidth="true">
      <ng-template bbDropdownLabel>
        <div role="button"
          data-role="context-navigation"
          class="user-context-dropdown__selector">
          <bb-user-context-menu-info
            [userInfo]="userInfo"
            [currentContext]="currentContext"
          ></bb-user-context-menu-info>
          <div class="user-context-dropdown__toggle">
            <bb-icon-ui name="toggle-down"></bb-icon-ui>
          </div>
        </div>
      </ng-template>
      <ng-template bbDropdownMenuItem>
        <bb-user-context-menu-dropdown
          [currentContext]="currentContext"
          [canSelectContext]="canSelectContext"
          (changeContext)="hostRef.openSwitchContextModal()"
          (logout)="hostRef.logout()"
        ></bb-user-context-menu-dropdown>
      </ng-template>
    </bb-dropdown-menu-ui>
  </div>

  <bb-modal-ui [(isOpen)]="hostRef.showSwitchContextModal">
    <bb-modal-body-ui>
      <ng-template bbCustomModalBody>
        <bb-select-context-widget *ngIf="hostRef.renderContextSelect"></bb-select-context-widget>
      </ng-template>
    </bb-modal-body-ui>
    <bb-modal-footer-ui>
      <ng-template bbCustomModalFooter>
        <button bbButton
          (click)="hostRef.showSwitchContextModal = false"
          color="secondary"
          data-role="cancel-button"
          i18n="Cancel button text@@user-context-menu.label.cancelButton">
          Cancel
        </button>
      </ng-template>
    </bb-modal-footer-ui>
  </bb-modal-ui>
</ng-template>

bbUserContextMenuDropdownCustomizable

This extension slot includes the following default template:

<ng-template
  bbUserContextMenuDropdownCustomizable
  let-hostRef
  let-currentContext="context.currentContext"
  let-canSelectContext="context.canSelectContext">

  <button
    class="dropdown-item bb-stack"
    role="menuitem"
    data-role="user-context-menu-dropdown-change-context-btn"
    (click)="hostRef.onChangeContext()"
  >
    <div class="bb-stack__item">
      <bb-user-context-icon-ui
        [isMaster]="currentContext?.isMaster"
        [name]="currentContext?.name"
      ></bb-user-context-icon-ui>
    </div>
    <div class="bb-stack__item">
      <div class="bb-text-bold" data-role="user-context">
        {{currentContext?.name}}
      </div>
      <div class="bb-text-support"
        data-role="change-context"
        *ngIf="canSelectContext"
        i18n="Change context|Click to change context label@@user-context-menu.label.clickToChange"
      >
        (click to change context)
      </div>
    </div>
  </button>

  <div class="dropdown-divider"></div>

  <button
    role="menuitem"
    class="dropdown-item"
    data-role="logout"
    (click)="hostRef.onLogout()"
  >
    <div
      class="user-context-menu-info"
      i18n="Log out|Click to log out label@@user-context-menu.label.logout"
    >
      Log out
    </div>
  </button>
</ng-template>

bbUserContextMenuInfoCustomizable

This extension slot includes the following default template:

<ng-template
  bbUserContextMenuInfoCustomizable
  let-hostRef
  let-userInfo="context.userInfo"
  let-currentContext="context.currentContext">
  <div class="user-context-menu-info">
    <bb-avatar-ui
      *ngIf="userInfo"
      [name]="userInfo?.fullName"
      class="user-context-menu-info__avatar"
    ></bb-avatar-ui>
    <div class="user-context-menu-info__description">
      <div *ngIf="userInfo"
        class="user-context-menu-info__name"
        data-role="user-full-name">
        {{userInfo?.fullName}}
      </div>
      <div class="user-context-menu-info__context" data-role="current-context">
        {{currentContext?.name}}
      </div>
    </div>
  </div>
</ng-template>
