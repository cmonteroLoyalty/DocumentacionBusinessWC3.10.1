Manage payables

Shared library

All Cash Flow Forecasting widgets use a shared library, see Cash Flow Forecasting common library.

Selector name

This widget has the following selector name: bb-cash-flow-cash-out-widget-ang

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    Import the following modules into the module of your wrapper widget:

    import { CashFlowCashOutWidgetAngModule } from '@backbase/business-ang/cash-flow';

  • Data module imports

    Import the following data modules into the module of your wrapper widget:

    import { CashFlowDataHttpModule } from '@backbase/data-ang/cash-flow';

Extension slots

This widget has the following extension slots:

  1. bbCashFlowCashOutWidgetCustomizable

  2. bbCashFlowCashOutTableCustomizable

  3. bbCashFlowCashOutTableItemCustomizable

  4. bbCashFlowCashOutListCustomizable

  5. bbCashFlowCashOutListItemCustomizable

bbCashFlowCashOutWidgetCustomizable

This extension slot includes the following default template:

<ng-template
  bbCashFlowCashOutWidgetCustomizable
  let-hostRef
  let-isLoading="context.isLoading"
  let-isError="context.isError"
  let-payables="context.payables"
  let-totalCount="context.totalCount"
  let-currentPage="context.currentPage"
  let-pageSize="context.pageSize"
  let-hasMorePayablesToLoad="context.hasMorePayablesToLoad"
  let-searchParamsAvailable="context.searchParamsAvailable"
  let-filter="context.filter"
  let-filters="context.filters"
  let-isMobileView="context.isMobileView"
  let-searchQuery="context.searchQuery"
  let-syncDatetime="context.syncDatetime"
  let-isFilterPanelAvailable="context.isFilterPanelAvailable"
>
  <bb-cash-flow-main-container
    [isErrorState]="isError"
    [isLoadingState]="isLoading"
    [isEmptyState]="!payables?.length"
    [isFilterPanelAvailable]="isFilterPanelAvailable"
  >
    <ng-container cash-flow-filter-controls>
      <div class="bb-cash-flow-search-params-container">
        <div class="bb-cash-flow-search-params-container__search">
          <bb-search-box-ui
            [showClear]="true"
            [showSearch]="true"
            i18n-placeholder="Search placeholder@@cash-out.search.placeholder"
            placeholder="Search"
            data-role="search-field"
            [ngModel]="searchQuery"
            (submit)="hostRef.onSubmit($event)"
            (clear)="hostRef.onClearSearch()"
          >
          </bb-search-box-ui>
        </div>
        <div
          class="bb-cash-flow-search-params-container__filter-and-time-container"
        >
          <bb-cash-flow-last-updated-time
            [dateTimestamp]="syncDatetime"
            wrapperClasses="bb-cash-flow-search-params-container__time-of-last-update"
          >
          </bb-cash-flow-last-updated-time>
          <div class="bb-cash-flow-search-params-container__filter">
            <bb-cash-flow-button-filter
              [activeFilter]="filter"
              [filters]="filters"
              (filterChanges)="hostRef.updateFilter($event)"
            >
            </bb-cash-flow-button-filter>
          </div>
        </div>
      </div>
      <p
        *ngIf="searchParamsAvailable && !isLoading && totalCount"
        data-role="bb-cash-flow-cash-out-payable__found-number"
        i18n="Total results found@@cash-out.search.search.total-count"
      >
        {totalCount, plural, =1 {{{ totalCount }} result found} other
        {{{ totalCount }} results found}}
      </p>
    </ng-container>
    <ng-container cash-flow-widget-body>
      <ng-container *ngIf="isMobileView; else tableView">
        <div class="bb-block" [class.bb-block--lg]="hasMorePayablesToLoad">
          <bb-cash-flow-cash-out-list
            [payables]="payables"
            (itemSelected)="hostRef.onItemSelected($event)"
          ></bb-cash-flow-cash-out-list>
        </div>
        <div class="bb-stack bb-stack--center">
          <bb-load-button-ui
            *ngIf="hasMorePayablesToLoad"
            [from]="currentPage"
            [size]="pageSize"
            [totalCount]="totalCount"
            [isLoading]="isLoading"
            (paramChange)="hostRef.onLoadMore(currentPage)"
            i18n="Load more button@@cash-out.table.load-more"
            color="secondary"
            data-role="bb-cash-flow-cash-out-payable__load-more-btn"
          >
            Load more
          </bb-load-button-ui>
        </div>
      </ng-container>
      <ng-template #tableView>
        <bb-cash-flow-cash-out-table
          [items]="payables"
          (itemSelected)="hostRef.onItemSelected($event)"
        ></bb-cash-flow-cash-out-table>
        <bb-pagination-ui
          [page]="currentPage"
          [boundaryLinks]="true"
          [directionLinks]="true"
          [totalItems]="totalCount"
          [itemsPerPage]="pageSize"
          [maxNavPages]="4"
          (pageChange)="hostRef.onPageChange($event)"
        >
        </bb-pagination-ui>
      </ng-template>
    </ng-container>
  </bb-cash-flow-main-container>
</ng-template>

This extension slot consumes the following object properties from its custom context:

Object property Description

isLoading: boolean

Whether the payables are loading.

isError: boolean

Whether there is an error preventing the widget to load the payables.

payables: CashFlowItem[]

The list of payables.

totalCount: number

Total number of payables.

currentPage: number

The current page number.

pageSize: number

Number of items on the page.

hasMorePayablesToLoad: boolean

Whether there are more payables to load.

searchParamsAvailable: boolean

Whether search parameters are available.

filter: PayablesFilter

filters: PayablesFilter[]

The currently active filters.

isMobileView: boolean

Whether mobile view is currently active.

searchQuery: string

The search query.

syncDatetime: string

Timestamp of the last data synchronization with the accounting provider.

isFilterPanelAvailable: boolean

Whether the filter panel exists.
