Manage batch file uploads

Properties

This widget has the following properties:

Label Name Admin Value type Default value

Number of batches per page

pageSize

X

long

50

Max. number of navigation pages

maxNavPages

X

long

3
Untitled

Select the method for loading more batches

How to load more data. The possible values are:

  • pagination: Click next page to load more data.

  • load-more: Scroll down to load more data.

loadMoreMethod

X

string

pagination

Access Control

This widget has no access control, as Manage batch file uploads is a personal list that contains files that only the current user has uploaded.

bbBatchLogCustomizable

This extension slot includes the following default template:

<ng-template bbBatchLogCustomizable let-hostRef>
  <ng-container *ngIf="!hostRef.error; else errorState">
    <bb-batch-log-list
      *ngIf="hostRef.batchUploads | async as batchUploads; else loadingLog"
      [batchUploads]="batchUploads.items"
      [loadMoreMethod]="hostRef.loadMoreMethod$ | async"
      [batchUploadsPerPage]="hostRef.pageSize$ | async"
      [totalBatchUploads]="batchUploads.totalCount"
      [loadMoreFrom]="hostRef.loadMoreFrom"
      [isLoading]="hostRef.isLoadingBatchUploads"
      [maxNavPages]="hostRef.maxNavPages$ | async"
      (pageChange)="hostRef.onPageChange($event)"
      (loadMore)="hostRef.onLoadMore($event)"
      (showDetails)="hostRef.onShowDetails($event)"
    ></bb-batch-log-list>
    <bb-batch-upload-details
      *ngIf="hostRef.selectedBatchUpload | async"
      [batchUpload]="hostRef.selectedBatchUpload | async"
      [isOpen]="hostRef.areDetailsOpen"
      [loading]="hostRef.isLoadingBatchUploadsDetails"
      [error]="hostRef.errorBatchUploadDetails"
      (close)="hostRef.onCloseDetails()"
      (submitDuplicateBatch)="hostRef.onSubmitDuplicateBatch($event)"
      (deleteDuplicateBatch)="hostRef.onDeleteDuplicateBatch($event)"
    ></bb-batch-upload-details>
  </ng-container>

  <ng-template #errorState>
    <div class="bb-state-container">
      <bb-empty-state-ui
        i18n-title="@@batch-log.error.generic.title"
        title="Page couldn't load"
        i18n-subtitle="@@batch-log.error.generic.subtitle"
        subtitle="Try to reload the page or contact the system administrator."
        iconSize="xxl"
        iconModifier="error"
      ></bb-empty-state-ui>
    </div>
  </ng-template>

  <ng-template #loadingLog>
    <div class="bb-state-container">
      <bb-loading-indicator-ui text="Loading data..."></bb-loading-indicator-ui>
    </div>
  </ng-template>

  <ng-template #updateDuplicateNotificationHeader let-status="status" let-action="action">
    <span data-role="duplicate-notification-header">
      <ng-container i18n="@@batch-log.notification.duplicate.title.success" *ngIf="status === 'SUCCESS'">
        Batch file {action, select, DELETE {deleted} other {processed}} successfully.
      </ng-container>
      <ng-container i18n="@@batch-log.notification.duplicate.title.failure" *ngIf="status === 'FAIL'">
        Batch file {action, select, DELETE {deletion} other {submission}} failed.
      </ng-container>
    </span>
  </ng-template>
  <ng-template #updateDuplicateNotification let-status="status" let-action="action">
    <span data-role="duplicate-notification-desc">
      <ng-container *ngIf="status === 'SUCCESS'" i18n="@@batch-log.notification.update-duplicate.body.success">
        The duplicate file was {action, select, DELETE {deleted} other {processed}}.
      </ng-container>
      <ng-container *ngIf="status === 'FAIL'" i18n="@@batch-log.notification.update-duplicate.body.failed">
        Unable to {action, select, DELETE {delete} other {submit}} duplicate file.
      </ng-container>
    </span>
  </ng-template>
</ng-template>
